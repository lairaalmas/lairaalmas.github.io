Criando um jogo novo

Para desenvolver um jogo novo para o iFractions, siga os proximos passos.

  1. Criar um arquivo novo em /js/games/nomeDoJogo.js contendo: 

    2. const nomeDoJogo = {

  /**
    * Main code
    */
  create: function () {
    if (moodle) {
      navigationIcons.add(false, false, false, 
      true, false, 
      false, false);
    } else {
      navigationIcons.add(true, true, false, 
      true, false, 
      'customMenu', false);
    }
    // ...
    game.timer.start(); // Set a timer for the current level (used in postScore)
    game.event.add('click', this.onInputDown);
    game.event.add('mousemove', this.onInputOver);
  },

  /**
    * Game loop
    */
  update: function () {}

  /**
    * Called by mouse click event
    *
    * @param {object} mouseEvent contains the mouse click coordinates
    */
  onInputDown: function (mouseEvent) {
    const shadowBlur = mouseEvent.offsetX, y = mouseEvent.offsetY;
    // ...
    navigationIcons.onInputDown(shadowBlur, y);
    game.render.all();
  },

  /**
    * Called by mouse move event
    *
    * @param {object} mouseEvent contains the mouse move coordinates
    */
  onInputOver: function (mouseEvent) {
    const shadowBlur = mouseEvent.offsetX, y = mouseEvent.offsetY;
    // ...
    navigationIcons.onInputOver(shadowBlur, y);
    game.render.all();
  },

}

  2. Adicionar o arquivo criado no <body> em index.html 

    3. <script src="js/games/nomeDoJogo.js"></script>

  3. Adicionar o novo estado do jogo no <script> em index.html 

    4. game.state.add('nomeDoJogo', nomeDoJogo);

  4. Colocar as informações sobre o jogo novo em globals.js 

    5. const info = {
  all: {
    nomeDoJogo: {
      // colocar aqui as propriedades
    },
  }
}

  5. Colocar as informações sobre as novas imagens em globals.js 

    6. const url: {

  boot: {
    image: [
      ['game<id>', <url>],
    ],
    sprite: [
      ['mode<id>', <url>],
    ]
  },
    
  nomeDoJogo: {
    image: [
      ['posiçãoInicialNoMapa', <url>],
      ['posiçãoFinalNoMapa', <url>],
    ],
    sprite: [
      ['personagemNoMapa', <url>],
    ],
    audio: [],
  }
  
}

  6. Atualizar assets

    7. Criar as novas figuras em:

    • /assets/img/levels/nomeDoJogo.png -> ícone de jogo na tela de menu principal.
    • /assets/img/levels/nomeDoJogo_1.png -> ícone de modo de jogo na tela de menu de customização.
